# Puppet Mania - Puppet Tool

scripted_gui = {
	pt_button = {
		window_name = "pt_button_container"
		parent_window_token = top_bar

		context_type = player_context

		visible = {
			ROOT = {
				is_ai = no
				is_subject = no
			}
		}

		effects = {
			pt_button_open_click = {
				set_variable = { show_pm_pt = 1 }
			}
			pt_button_close_click = {
				clear_variable = show_pm_pt
				clear_variable = show_pm_pt_dropdown
				clear_variable = pt_puppet_target
				every_state = {
					limit = {
						ROOT = {
							is_in_array = {
								array = pt_transfer_states_ar
								value = PREV.id
							}
						}
					}
					ROOT = {
						remove_from_array = {
							array = pt_transfer_states_ar
							value = PREV.id	
						}
					}
				}
			}
		}

		triggers = {
			pt_button_open_visible = {
				NOT = {
					has_variable = show_pm_pt
				}
			}
			pt_button_close_visible = {
				has_variable = show_pm_pt
			}
			puppet_off_visible = {
				NOT = {
					has_variable = show_pm_pt
				}
			}
			puppet_on_visible = {
				has_variable = show_pm_pt
			}
		}
	}

	pt_logic = {
		context_type = selected_state_context

		window_name = "pt_main_container"

		visible = {
			ROOT = {
				is_ai = no
				has_variable = show_pm_pt
			}
			FROM.FROM = {
				owner = {
					OR = {
						tag = ROOT
						is_subject_of = ROOT
					}
				}
				is_fully_controlled_by = owner
			}
		}	

		effects = {
			pt_state_marker_click = {
				FROM.FROM = {
					ROOT = {
						add_to_array = {
							array = pt_transfer_states_ar
							value = PREV.id
						}
					}
				}
			}
			pt_state_unmarker_click = {
				FROM.FROM = {
					ROOT = {
						remove_from_array = {
							array = pt_transfer_states_ar
							value = PREV.id
						}
					}
				}
			}
			pt_close_button_click = {
				ROOT = {
					clear_variable = show_pm_pt
					clear_variable = show_pm_pt_dropdown
					clear_variable = pt_puppet_target
				}
				every_state = {
					limit = {
						ROOT = {
							is_in_array = {
								array = pt_transfer_states_ar
								value = PREV.id
							}
						}
					}
					ROOT = {
						remove_from_array = {
							array = pt_transfer_states_ar
							value = PREV.id	
						}
					}
				}
			}
			pt_transfer_button_click = {
				if = {
					limit = { var:pt_puppet_target = { exists = yes } }
					ROOT = {
						create_dynamic_country = {
							original_tag = pt_puppet_target
							every_state = {
								limit = {
									is_fully_controlled_by = owner
									is_owned_by = ROOT
									ROOT = {
										is_in_array = {
											array = pt_transfer_states_ar
											value = PREV.id
										}
									}
								}
								transfer_state_to = PREV
								ROOT = {
									remove_from_array = {
										array = pt_transfer_states_ar
										value = PREV.id
									}
								}
							}
							ROOT = {
								# Puppeting
								if = {
									limit = {
										OR = {
											AND = {
												tag = JAP
												has_government = fascism
											}
											AND = {
												tag = MAN
												OR = {
													has_government = fascism
													has_government = neutrality
												}
											}
										}
									}
									set_autonomy = {
										target = PREV
										autonomy_state = autonomy_wtt_imperial_subject
									}
								}
								else_if = {
									limit = {
										has_government = fascism
										NOT = { tag = JAP }
									}
									set_autonomy = {
										target = PREV
										autonomy_state = autonomy_reichsprotectorate
									}
								}
								else = {
									set_autonomy = {
										target = PREV
										autonomy_state = autonomy_puppet
									}
								}
							}
						}
					}
				}
				else = {
					ROOT = {
						var:pt_puppet_target = {
							every_state = {
								limit = {
									is_fully_controlled_by = owner
									is_owned_by = ROOT
									ROOT = {
										is_in_array = {
											array = pt_transfer_states_ar
											value = PREV.id
										}
									}
								}
								transfer_state_to = PREV
								ROOT = {
									remove_from_array = {
										array = pt_transfer_states_ar
										value = PREV.id
									}
								}
							}
							ROOT = {
								# Puppeting
								if = {
									limit = {
										OR = {
											AND = {
												original_tag = JAP
												has_government = fascism
											}
											AND = {
												original_tag = MAN
												OR = {
													has_government = fascism
													has_government = neutrality
												}
											}
										}
									}
									set_autonomy = {
										target = PREV
										autonomy_state = autonomy_wtt_imperial_subject
									}
								}
								else_if = {
									limit = {
										has_government = fascism
										NOT = { original_tag = JAP original_tag = MAN }
									}
									set_autonomy = {
										target = PREV
										autonomy_state = autonomy_reichsprotectorate
									}
								}
								else = {
									set_autonomy = {
										target = PREV
										autonomy_state = autonomy_puppet
									}
								}
							}
						}
					}
				}
			}

			pt_drop_down_list_button_closed_click = {
				ROOT = {
					set_variable = { show_pm_pt_dropdown = 1 }
				}
			}
			pt_drop_down_list_button_opened_click = {
				ROOT = {
					clear_variable = show_pm_pt_dropdown
				}
			}
		}

		triggers = {
			pt_state_marker_visible = {
				FROM.FROM = {
					ROOT = {
						NOT = {
							is_in_array = {
								array = pt_transfer_states_ar
								value = PREV.id
							}
						}
						owns_state = PREV
					}
					is_fully_controlled_by = owner
				}
			}
			pt_state_unmarker_visible = {
				FROM.FROM = {
					ROOT = {
						is_in_array = {
							array = pt_transfer_states_ar
							value = PREV.id
						}
						owns_state = PREV
					}
					is_fully_controlled_by = owner
				}
			}
			pt_state_marker_blocked_visible = {
				NOT = {
					FROM.FROM = {
						is_fully_controlled_by = owner
						owner = {
							tag = ROOT
						}
					}
				}
			}
			pt_transfer_button_visible = {
				ROOT = {
					has_variable = pt_puppet_target
					any_owned_state = {
						is_fully_controlled_by = owner
						ROOT = {
							is_in_array = {
								array = pt_transfer_states_ar
								value = PREV.id
							}
						}
					}
					has_capitulated = no
					if = {
						limit = {
							has_game_rule = {
								rule = pm_pt_puppet_during_war_status
								option = pm_pt_no_puppet_during_war
							}
						}
						has_war = no
					}
					else_if = {
						limit = {
							has_war = yes
							has_game_rule = {
								rule = pm_pt_puppet_during_war_status
								option = pm_pt_puppet_during_war_low_cap
							}
						}
						surrender_progress < 0.25
					}
					else_if = {
						limit = {
							has_war = yes
							has_game_rule = {
								rule = pm_pt_puppet_during_war_status
								option = pm_pt_puppet_during_war_low_cap
							}
						}
						surrender_progress < 0.01
					}
				}
			}
			pt_transfer_button_cant_visible = {
				ROOT = {
					OR = {
						NOT = { has_variable = pt_puppet_target }
						NOT = {
							any_owned_state = {
								is_fully_controlled_by = owner	
								ROOT = {
									is_in_array = {
										array = pt_transfer_states_ar
										value = PREV.id
									}
								}
							}
						}
						has_capitulated = yes
						if = {
							limit = {
								has_game_rule = {
									rule = pm_pt_puppet_during_war_status
									option = pm_pt_no_puppet_during_war
								}
							}
							has_war = yes
						}
						else_if = {
							limit = {
								has_war = yes
								has_game_rule = {
									rule = pm_pt_puppet_during_war_status
									option = pm_pt_puppet_during_war_low_cap
								}
							}
							surrender_progress > 0.24
						}
						else_if = {
							limit = {
								has_war = yes
								has_game_rule = {
									rule = pm_pt_puppet_during_war_status
									option = pm_pt_puppet_during_war_low_cap
								}
							}
							surrender_progress > 0
						}
					}
				}
			}

			pt_drop_down_list_button_closed_visible = {
				ROOT = {
					NOT = { has_variable = show_pm_pt_dropdown }
				}
			}
			pt_drop_down_list_button_opened_visible = {
				ROOT = {
					has_variable = show_pm_pt_dropdown
				}
			}
			pt_drop_down_list_button_blocked_visible = {
				NOT = {
					any_country_with_core = {
						NOT = {
							tag = ROOT
						}
					}
				}
			}
		}

		dynamic_lists = {
			pt_state_gridbox = {
				array = ROOT.pt_transfer_states_ar

				value = state_id

				change_scope = yes

				entry_container = "pt_state_entry"
			}
		}
	}

	pt_dropdown_window = {
		window_name = "pt_drop_down_list"
		parent_scripted_gui = pt_logic

		context_type = selected_state_context

		visible = {
			ROOT = {
				is_ai = no
				has_variable = show_pm_pt_dropdown
			}
			any_country_with_core = {
				NOT = {
					tag = ROOT
				}
			}
		}

		effects = {
			pt_country_entry_select_button_click = {
				ROOT = {
					clear_variable = pt_puppet_target
					set_variable = {
						pt_puppet_target = core_country
					}
				}
			}
			pt_country_entry_deselect_button_click = {
				ROOT = {
					clear_variable = pt_puppet_target
				}
			}
		}

		triggers = {
			pt_country_entry_select_button_visible = {
				ROOT = {
					NOT = {
						var:pt_puppet_target = {
							tag = var:core_country
						}
						has_subject = var:core_country
					}
				}
			}
			pt_country_entry_deselect_button_visible = {
				ROOT = {
					var:pt_puppet_target = {
						tag = var:core_country
					}
					NOT = { has_subject = var:core_country }
				}
			}
			pt_country_entry_select_button_blocked_visible = {
				ROOT = {
					has_subject = var:core_country
				}
			}
			pt_country_entry_select_button_blocked_text_visible = {
				ROOT = {
					has_subject = var:core_country
				}
			}
		}

		dynamic_lists = {
			pt_country_gridbox = {
				array = core_countries

				value = core_country

				change_scope = yes

				entry_container = "[PT_COUNTRY_ENTRY_SL]"
			}
		}
	}
}
